# plays a music cue once per player character's lifetime
# example use : play_music_cue_once = { TRACK = "mx_cue_combat_stinger"	}
play_music_cue_once = {
	if = {
		limit = {
			is_ai = no
			NOT = {
				has_character_flag = played_$TRACK$
			}
		}
		add_character_flag = played_$TRACK$
		play_music_cue = $TRACK$
	}
}
