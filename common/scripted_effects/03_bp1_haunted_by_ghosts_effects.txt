# Scripted effect to determine what sin you have and what virtue you should strive for
bp1_yearly_9026_determine_sin_and_virtue = {
	if = {
		limit = {
			OR = {
				has_trait = greedy
				has_trait = gluttonous
				has_trait = chaste
			}
		}
		random_owned_story = {
			limit = {
				story_type = story_being_visited_by_angry_ghosts
			}
			set_variable = {
				name = trait_to_gain
				value = 1 #generous
			}
		}
	}
	else_if = {
		limit = {
			OR = {
				has_trait = murderer
				any_secret = {
					secret_type = secret_murder
				}
				has_trait = wrathful
				has_trait = vengeful
			}
		}
		random_owned_story = {
			limit = {
				story_type = story_being_visited_by_angry_ghosts
			}
			set_variable = {
				name = trait_to_gain
				value = 2 # forgiving
			}
		}
	}
	else = {
		random_owned_story = {
			limit = {
				story_type = story_being_visited_by_angry_ghosts
			}
			set_variable = {
				name = trait_to_gain
				value = 3 #compassionate
			}
		}
	}
}

# Increase good deeds and give a tooltip for it
bp1_increase_good_deeds = {
	custom_tooltip = {
		text = bp1_increase_good_deeds_tt
		random_owned_story = {
			limit = {
				story_type = story_being_visited_by_angry_ghosts
			}
			change_variable = {
				name = good_deeds
				add = 1
			}
		}
	}
}

# Increase good deeds and give a tooltip for it
bp1_decrease_good_deeds = {
	custom_tooltip = {
		text = bp1_decrease_good_deeds_tt
		random_owned_story = {
			limit = {
				story_type = story_being_visited_by_angry_ghosts
			}
			change_variable = {
				name = good_deeds
				subtract = 1
			}
		}
	}
}
